[attr.x]="y"

Set attribute x on element to value y.

component code
image = 'assets/sad.png';
makeHappy() {
this.image = 'assets/happy.png';
}

component template
<img [attr.src]="image" />
<button (click)="makeHappy()">Smile</button>
output

[class.x]="y"

If y == true, add class x to the element.

component code
hasBorder = false;
addBorder() {
this.hasBorder = true;
}

component template
<div [class.border]="hasBorder">Lorem Ipsum</div>
<button (click)="addBorder()">Add border</button>
output
Lorem Ipsum

[ngClass]

More powerful ways for setting classes

component code
myNumber = 1;
getMyClass() {
return {
  'text-muted': this.myNumber == 1,
  'text-primary': this.myNumber == 2,
  'text-success': this.myNumber == 3,
  'text-info': this.myNumber == 4,
  'text-warning': this.myNumber == 5,
  'text-danger': this.myNumber == 6
};
}

component template
<div [ngClass]="getMyClass()">Lorem Ipsum</div>
<input type="range" min="1" max="6" [(ngModel)]="myNumber" />
output
Lorem Ipsum

[NgStyle]

More powerful ways for setting styles

component code
width = 40;
getStyle() {
return {
  'width.px': this.width,
  'background-color': this.width > 80 ? 'red' : 'lightgreen'
};
}

component template
<div [ngStyle]="getStyle()">Lorem Ipsum</div>
<input type="range" min="40" max="100" [(ngModel)]="width" />
{{width}}
output
Hello!
{{width}}
